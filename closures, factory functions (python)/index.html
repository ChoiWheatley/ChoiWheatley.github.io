    def in_cache(func):
    cache = {}

    def wrapper(n):
        print(f"DEBUG ---> n: {n}, cache: {cache}")
        if n in cache:
            print("DEBUG ---> CACHE HIT")
            return cache[n]
        else:
            print("DEBUG ---> CACHE MISS")
            cache[n] = func(n)
            return cache[n]

    return wrapper


def factorial(n):
    ret = 1
    for i in range(1, n + 1):
        ret *= i
    return ret


def square(n):
    return n**2
